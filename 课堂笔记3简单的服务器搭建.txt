2.编写gulpfile.js配置文件
var gulp=require('gulp');
var webserver=require('gulp-webserver');

gulp.task('webserver',function(){
	gulp.src('./')
		.pipe(webserver({
			port:8080,
			host:'localhost',
			livereload:true,
			directoryListing:true,
			open:true
		}))
})
		
3.运用web服务器
